library(Seurat)
library(here)
dir.create(here('bcells'))
'/home/rstudio/tabula-muris-vignettes/bcells' already exists
source('~/tabula-muris/00_data_ingest/02_tissue_analysis_rmd/boilerplate.R')
robjs = c('~/tabula-muris/00_data_ingest/11_global_robj/FACS_all.Robj',
          '~/tabula-muris/00_data_ingest/11_global_robj/droplet_all.Robj')
for (robj in robjs){
  robj_basename = basename(robj)
  filename = sub('.robj', '_bcells.robj', here('bcells', robj_basename))
  
  # load(robj)
  if (length(grep('droplet', robj_basename)) > 0){
    tiss = tiss_droplet
    scale = 1e4
  } else {
    tiss = tiss_FACS
    scale = 1e6
  }
  bcell_ind = grepl('(B cell)|(leukocyte)', tiss@meta.data$cell_ontology_class, perl=TRUE)
  bcell_ontologies = unique(tiss@meta.data$cell_ontology_class[bcell_ind])
  print(bcell_ontologies)
  bcell_ids = tiss@cell.names[bcell_ind]
  tiss_bcells = SubsetData(tiss, cells.use=bcell_ids)
  tiss_bcells = process_tissue(tiss_bcells, scale=scale)
  filename = sub('.robj', '_bcells.robj', here('bcells', robj_basename))
  print(filename)
  if (length(grep('droplet', robj_basename)) > 0){
    tiss_droplet_bcells = tiss_bcells
    save(tiss_droplet_bcells, file=filename)
  } else {
    tiss_facs_bcells = tiss_bcells
    save(tiss_facs_bcells, file=filename)
  }
}
